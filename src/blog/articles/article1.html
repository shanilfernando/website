A Cross-browser solution to access CSS transition in javascript

Januray 3, 2018

Issue :  The transition return empty in some browsers.

Solution :  Access each transition property seperatly as follows. 

  1. function getCSSStyles(el) {
  2.  var style = getComputedStyle(el, null);
  3.  var result = {};
  4.  
  5.  for (var _len = arguments.length, properties = Array(_len > 1 ? _len - 1 : 0), _key = 1; _key < _len; _key++) {
  6.   properties[_key - 1] = arguments[_key];
  7.  }
  8.  
  9.  properties.forEach(function(value) {
  10.   var property = style.getPropertyValue('-webkit-' + value) || style.getPropertyValue('-moz-' + value) || style.getPropertyValue('-ms-' + value) || style.getPropertyValue('-o-' + value) || style.getPropertyValue('' + value);
  11.   result[value] = property;
  12.  });
  13.  return result;
  14. }
  15.  
  16.  
  17. var xiuli = document.getElementById('xiuli');
  18. var trans = getCSSStyles(xiuli, 'transition-property', 'transition-duration', 'transition-timing-function', 'transition-delay');

If you want still needs the shorthand transition, change getCSSStyles function as follows 

  1. function getCSSStyles(el) {

  4.

  11.   result.push(property);
  12.  });
  13.  return result.join(' ');
  14. }

Explanation:  Even the CSS transition is defined as following, 

  1. #xiuli {
  2.       -webkit-transition: all 2s linear 1s;
  3.       -moz-transition: all 2s linear 1s;
  4.       -o-transition: all 2s linear 1s;
  5.       transition: all 2s linear 1s;
  6. }
  1. #xiuli {
  2.       -webkit-transition: all 2s linear 1s;
  3.       -moz-transition: all 2s linear 1s;
  4.       -o-transition: all 2s linear 1s;
  5.       transition: all 2s linear 1s;
  6. }

When to try to get transition using transition in you javascript code, you will endup with following results

  1. // chrome, safari
  2. "transition" : "all 2s linear 1s"
  3. "transition-delay" : "1s"
  4. "transition-duration" : "2s"
  5. "transition-property" : "all"
  6. "transition-timing-function" : "linear"
  7.  
  8. // firefox, opera
  9. "transition": ""
  10. "transition-delay": "1s"
  11. "transition-duration": "2s"
  12. "transition-property": "all"
  13. "transition-timing-function": "linear"

A workaround for this issue is to access each transition property seperatly.
